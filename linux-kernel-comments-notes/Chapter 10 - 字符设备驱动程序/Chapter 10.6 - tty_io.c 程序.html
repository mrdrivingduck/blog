    struct tty_struct * tty;

        return -EIO;

        (tty->pgrp != current->pgrp))

        return (tty_signal(SIGTTOU, tty));

        sleep_if_full(tty->write_q); // ÂÜôÈòüÂàóÂ∑≤Êª°

            // ÂΩìÂâçËøõÁ®ãÊúâ‰ø°Âè∑Ë¶ÅÂ§ÑÁêÜ

        tty->write(tty); // Ë∞ÉÁî®ÂØπÂ∫îÁöÑ tty ÂÜôÂáΩÊï∞
