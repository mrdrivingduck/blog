    while (argc-- > 0) {

            return 0;
        }

    }

}

int do_execve(unsigned long * eip, long tmp, char * filename, char ** argv, char ** envp)
{
    struct m_inode * inode;
    struct buffer_head * bh;
    unsigned long page[MAX_ARG_PAGES];
    int i, argc, envc;
    int e_uid, e_gid;
    int retval;
    int sh_bang = 0; // ÊòØÂê¶ÈúÄË¶ÅÊâßË°åËÑöÊú¨Á®ãÂ∫è

        panic("execve called from supervisor mode");
    for (i = 0; i < MAX_ARG_PAGES; i++)
        // È°µÈù¢Ê∏ÖÈõ∂
        page[i] = 0;
    if (!(inode = namei(filename))) // ÂèñÊâßË°åÊñá‰ª∂ÁöÑ inode
        return -ENOENT;
    argc = count(argv); // ËÆ°ÁÆóÂëΩ‰ª§Ë°åÂèÇÊï∞‰∏™Êï∞
    envc = count(envp); // ËÆ°ÁÆóÁéØÂ¢ÉÂ≠óÁ¨¶‰∏≤ÂèòÈáè‰∏™Êï∞
    
restart_interp:
    // Ë¶ÅÊâßË°åÁöÑÂøÖÈ°ªÊòØ‰∏Ä‰∏™Êñá‰ª∂
    if (!S_ISREG(inode->i_mode)) {
        retval = -EACCES;
        goto exec_error2;
    }
    
    // ÊùÉÈôêÂà§Êñ≠

    e_uid = (i & S_ISUID) ? inode->i_uid : current->euid;
    e_gid = (i & S_ISGID) ? inode->i_gid : current->egid;
    if (current->euid == inode->i_uid)
        i >>= 6;
    else if (in_group_p(inode->i_gid))
        i >>= 3;
    if (!(i & 1) && !((inode->i_mode & 0111) && suser())) {
        retval = -ENOEXEC;
        goto exec_error2;
    }
    
    // ÊùÉÈôêÊ£ÄÊü•ÈÄöËøá
    // ÂèñÂá∫Êñá‰ª∂È¶ñÈÉ®ÁöÑÊï∞ÊçÆ

    }
    ex = *((struct exec *) bh->b_data); // exec Â§¥ÈÉ®

        // Êñá‰ª∂Â§¥ÈÉ®‰ª• #! ÂºÄÂ§¥
        // ...

        ex.a_drsize ||

        goto exec_error2;

    if (!sh_bang) {

        p = copy_strings(envc, envp, page, p, 0);
        p = copy_strings(argc, argv, page, p, 0);
        if (!p) {
            retval = -ENOMEM;
            goto exec_error2;
        }
    }
    
    if (current->executable)
        iput(current->executable); // ÊîæÂõûÂéüÊâßË°åÁ®ãÂ∫èÁöÑ inode
    current->executable = inode; // ËÆæÁΩÆÊñ∞ÁöÑÊâßË°åÊñá‰ª∂ inode
    current->signal = 0; // Â§ç‰ΩçÊâÄÊúâ‰ø°Âè∑

    for (i = 0; i < 32; i++) {

        current->sigaction[i].sa_flags = 0;
        if (current->sigaction[i].sa_handler != SIG_IGN)

    }
    // ÂÖ≥Èó≠ÊâìÂºÄÁöÑÊñá‰ª∂ (Ê†πÊçÆ close_on_exec Ê†áÂøó)
    for (i = 0; i < NR_OPEN; i++)
        if ((current->close_on_exec >> i) & 1)
            sys_close(i);
    current->close_on_exec = 0;
    
    // ÈáäÊîæÂéüÊù•Á®ãÂ∫èÁöÑ‰ª£Á†ÅÊÆµÂíåÊï∞ÊçÆÊÆµÊâÄÂØπÂ∫îÁöÑÁâ©ÁêÜÈ°µÈù¢ÂíåÈ°µË°®

    free_page_tables(get_base(current->ldt[2]), get_limit(0x17));
    // Â¶ÇÊûúÂéüËøõÁ®ã‰ΩøÁî®‰∫ÜÂçèÂ§ÑÁêÜÂô®
    if (last_task_used_math == current)

    current->used_math = 0;
    // Âú®Ê†àÁ©∫Èó¥‰∏≠ÂàõÂª∫ÊåáÈíàË°®ÔºåÂÖ± main() ‰ΩøÁî®
    p += change_ldt(ex.a_text, page);
    p -= LIBRARY_SIZE + MAX_ARG_PAGES * PAGE_SIZE;
    p = (unsigned long) create_tables((char *) p, argc, envc);
    
    // ‰øÆÊîπËøõÁ®ã PCB

        (current->end_data = ex.a_data +
         (current->end_code = ex.a_text));

    current->suid = current->euid = e_uid;
    current->sgid = current->egid = e_gid;
    
    eip[0] = ex.a_entry; // eip ÊåáÂêëÁ®ãÂ∫èÂÖ•Âè£Âú∞ÂùÄ
    eip[3] = p; // esp ÊåáÂêëÂèÇÊï∞Ë°®È°∂Á´Ø
    return 0;
    
exec_error2:

exec_error1:

        free_page(page[i]);
    return (retval);
}
