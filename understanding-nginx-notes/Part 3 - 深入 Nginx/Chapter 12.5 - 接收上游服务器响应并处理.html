ngx_int_t
ngx_http_upstream_non_buffered_filter(void *data, ssize_t bytes)
{
    ngx_http_request_t  *r = data;

    ngx_buf_t            *b;
    ngx_chain_t          *cl, **ll;
    ngx_http_upstream_t  *u;

    u = r->upstream;

    // æ‰¾åˆ° out_bufs çš„æœ«å°¾ä»¥ä¾¿æ·»åŠ æ–°çš„ç¼“å†²åŒº
    for (cl = u->out_bufs, ll = &u->out_bufs; cl; cl = cl->next) {
        ll = &cl->next;
    }

    // è·å¾—ä¸€ä¸ªç©ºé—²çš„ ngx_buf_t
    cl = ngx_chain_get_free_buf(r->pool, &u->free_bufs);
    if (cl == NULL) {
        return NGX_ERROR;
    }

    // æ–°çš„ç¼“å†²å—åŠ å…¥é“¾è¡¨å°¾
    *ll = cl;

    cl->buf->flush = 1;
    cl->buf->memory = 1;

    b = &u->buffer;

    // Buffer æŒ‡é’ˆå‘åç§»åŠ¨ï¼Œä¿å­˜ body
    cl->buf->pos = b->last;
    b->last += bytes;
    cl->buf->last = b->last;
    cl->buf->tag = u->output.tag;

    if (u->length == -1) {
        return NGX_OK;
    }

    if (bytes > u->length) {

        ngx_log_error(NGX_LOG_WARN, r->connection->log, 0,
                      "upstream sent more data than specified in "
                      "\"Content-Length\" header");

        cl->buf->last = cl->buf->pos + u->length;
        u->length = 0;

        return NGX_OK;
    }

    // éœ€è¦æ¥æ”¶çš„ body é•¿åº¦å‡å°‘
    u->length -= bytes;

    return NGX_OK;
}
