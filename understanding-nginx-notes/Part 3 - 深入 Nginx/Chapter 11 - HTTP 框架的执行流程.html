    port = c->listening->servers;

    if (port->naddrs > 1) {

            addr6 = port->addrs;

                    break;

                    break;

            break;

            break;

            hc->addr_conf = &addr[0].conf;
            break;

#if (NGX_HTTP_SSL)

        }
    }

    ngx_http_run_posted_requests(c);
}

            h->key.len = r->header_name_end - r->header_name_start;
            h->key.data = r->header_name_start;
            h->key.data[h->key.len] = '\0';
